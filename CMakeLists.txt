cmake_minimum_required(VERSION 2.6)

PROJECT(CompetitionED3D)

FIND_PACKAGE(OpenThreads)
FIND_PACKAGE(OpenGL)
FIND_PACKAGE(osg)
FIND_PACKAGE(osgDB)
FIND_PACKAGE(osgViewer)
FIND_PACKAGE(osgGA)
FIND_PACKAGE(osgUtil)

SET(HEADERS 
   src/Scanner.h
   src/ScreenCapture.h
)

SET(SOURCES
src/main.cpp
src/Scanner.cpp
src/ScreenCapture.cpp
)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")

INCLUDE_DIRECTORIES(${OPENTHREADS_INCLUDE_DIR} ${OSG_INCLUDE_DIR} ${CMAKE_SOURCE_DIR})

LINK_DIRECTORIES(${OSG_LIB_DIR})

ADD_EXECUTABLE(main ${SOURCES})

TARGET_LINK_LIBRARIES(main ${OSG_LIBRARY} ${OSGDB_LIBRARY} ${OSGVIEWER_LIBRARY} ${OSGGA_LIBRARY} ${OSGUTIL_LIBRARY} ${OPENTHREADS_LIBRARY} ${OPENGL_LIBRARY})
